206void MyController::execDelayMs(uint32_t ms) {
207
208 // Anzahl der Mikrosekunden basierend auf den Millisekunden berechnen
209

216 if (ms < (maxMillis)) {

222

224 uint8_t numberOfLoops = (us / sizeOf16Bit);

233 // Verzögere den verbleibenden Teil
