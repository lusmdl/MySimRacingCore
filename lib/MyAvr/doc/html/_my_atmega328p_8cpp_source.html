177void MyAtmega328p ::initUart(uint32_t baudrate) {
178
179 uint16_t ubrr = F_CPU / (16 * baudrate) - 1;
180
181 // Set baud rate
182
183 UBRR0H = (uint8_t)(ubrr >> 8);
184 UBRR0L = (uint8_t)ubrr;
185
186 // Enable receiver and transmitter
187

191
192 // UCSR0C = (1 << UCSZ01) | (1 << UCSZ00);
193 UCSR0C = (1 << USBS0) | (3 << UCSZ00);
194
195 printUart("\nHello World!\n");

454void MyAtmega328p::execSleep(enum_sleepmodes mode, uint8_t bits_power_reduction, bool enable_bod_sleep) {
455
456 // Disable interrupts
457 cli();
458
459 // Disable sleep mode
460 setBit(MCUCR, SE, false);
461
462 // Set the requested sleep mode
463 switch(mode) {
464
465 case IDLE:
466
467 clearBitMask(SMCR, BITMASK(SM0) | BITMASK(SM1) | BITMASK(SM2));
468
469 break;
470
471 case ADC_NOISE_REDUCTION:
472
473 setBitMask(SMCR, BITMASK(SM0));
474 clearBitMask(SMCR, BITMASK(SM1) | BITMASK(SM2));
475
476 break;
477
478 case POWER_DOWN:
479
480 setBitMask(SMCR, BITMASK(SM1));
481 clearBitMask(SMCR, BITMASK(SM2) | BITMASK(SM0));
482
483 break;
484
485 case POWER_SAVE:
486
487 setBitMask(SMCR, BITMASK(SM0) | BITMASK(SM1));
488 clearBitMask(SMCR, BITMASK(SM2));
489
490 break;
491
492 case STANDBY:
493
494 // only recommend for use with external crystals or resonators
495
496 clearBitMask(SMCR, BITMASK(SM0));
497 setBitMask(SMCR, BITMASK(SM1) | BITMASK(SM2));
498
499 break;
500
501 case EXTENDED_STANDBY:
502
503 // only recommend for use with external crystals or resonators
504
505 setBitMask(SMCR, BITMASK(SM0) | BITMASK(SM1) | BITMASK(SM2) );
506
507 break;
508
509 default:
510
511 // Invalid sleep mode, return imidiately
512
513 return;
514 }
515
516 // Set the power reduction bits
517 PRR |= bits_power_reduction;
518
519 if (enable_bod_sleep) {
520
521 // Enable BODS bit and set BODS to disable BOD in sleep mode if enableBODSleep is true
522
523 setBit(MCUCR, BODS);
524 setBit(MCUCR, BODSE);
525 clearBit(MCUCR, BODSE);
526 }
527
528 // Delay for three clock cycles
529
530 asm volatile("nop");
531 asm volatile("nop");
532 asm volatile("nop");
533
534 // Enable sleep mode
535 setBit(SMCR, SE);
536
537 // Enable interrupts
538 sei();
539
540 // Enter sleep mode
541 asm volatile("sleep");
542
543 // Execution will resume here after waking up
544 //...
545
546 // Clear the power reduction bits
547 PRR &= ~bits_power_reduction;
548
549 if (enable_bod_sleep) {
550
551 // Re-enable BOD after sleep mode if enableBODSleep is true (optional)
552
553 clearBit(MCUCR, BODS);
554 }
555
556 // Disable sleep mode
557
558 setBit(MCUCR, SE, false);
559 clearBitMask(SMCR, BITMASK(SM0) | BITMASK(SM1) | BITMASK(SM2));
560}
